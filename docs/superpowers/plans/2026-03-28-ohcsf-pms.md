# OHCSF PMS Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Extend the People/HR Performance module with Nigeria's OHCSF Performance Management System, gated by a `pms_ohcsf_enabled` feature flag on the Organization model.

**Architecture:** Feature-flagged extension of existing perf module. New models in `app/models/people/perf/`, new services in `app/services/people/perf/`, new routes under `/people/perf/pms/`. All new tables in the `perf` schema. Three Alembic migrations: extend existing, create new tables, seed data.

**Tech Stack:** SQLAlchemy 2.0, FastAPI, Jinja2/HTMX/Alpine.js, Celery, PostgreSQL, pytest

**Spec:** `docs/superpowers/specs/2026-03-28-ohcsf-pms-design.md`

# app/models/people/perf/performance_contract.py
"""Performance Contract model — OHCSF PMS."""

import uuid
from datetime import date, datetime

from sqlalchemy import (
    Date,
    DateTime,
    Enum,
    ForeignKey,
    Index,
    String,
    Text,
    UniqueConstraint,
    func,
    text,
)
from sqlalchemy.dialects.postgresql import JSON, UUID
from sqlalchemy.orm import Mapped, mapped_column, relationship

from app.db import Base
from app.models.people.base import AuditMixin
from app.models.people.perf.pms_enums import ContractStatus, ContractType


class PerformanceContract(Base, AuditMixin):
    """Formal signed performance agreement (PPF/Performance Contract)."""

    __tablename__ = "performance_contract"
    __table_args__ = (
        UniqueConstraint(
            "organization_id", "contract_code", name="uq_perf_contract_code"
        ),
        Index("idx_contract_employee", "employee_id"),
        Index("idx_contract_cycle", "cycle_id"),
        Index("idx_contract_org_status", "organization_id", "status"),
        {"schema": "perf"},
    )

    contract_id: Mapped[uuid.UUID] = mapped_column(
        UUID(as_uuid=True),
        primary_key=True,
        default=uuid.uuid4,
        server_default=text("gen_random_uuid()"),
    )
    organization_id: Mapped[uuid.UUID] = mapped_column(
        UUID(as_uuid=True),
        ForeignKey("core_org.organization.organization_id"),
        nullable=False,
    )
    cycle_id: Mapped[uuid.UUID] = mapped_column(
        UUID(as_uuid=True),
        ForeignKey("perf.appraisal_cycle.cycle_id"),
        nullable=False,
    )
    employee_id: Mapped[uuid.UUID] = mapped_column(
        UUID(as_uuid=True),
        ForeignKey("hr.employee.employee_id"),
        nullable=False,
    )
    supervisor_id: Mapped[uuid.UUID] = mapped_column(
        UUID(as_uuid=True),
        ForeignKey("hr.employee.employee_id"),
        nullable=False,
    )
    contract_code: Mapped[str] = mapped_column(String(30), nullable=False)
    contract_type: Mapped[ContractType] = mapped_column(
        Enum(ContractType, name="contract_type", schema="perf"),
        nullable=False,
    )
    status: Mapped[ContractStatus] = mapped_column(
        Enum(ContractStatus, name="contract_status", schema="perf"),
        default=ContractStatus.DRAFT,
    )

    # Content
    objectives: Mapped[list] = mapped_column(JSON, nullable=False)
    competency_ids: Mapped[list | None] = mapped_column(JSON, nullable=True)
    development_plan: Mapped[str | None] = mapped_column(Text, nullable=True)

    # Signatures
    employee_signed_date: Mapped[date | None] = mapped_column(Date, nullable=True)
    supervisor_signed_date: Mapped[date | None] = mapped_column(Date, nullable=True)
    countersigner_id: Mapped[uuid.UUID | None] = mapped_column(
        UUID(as_uuid=True),
        ForeignKey("hr.employee.employee_id"),
        nullable=True,
    )
    countersigner_date: Mapped[date | None] = mapped_column(Date, nullable=True)

    # Amendments
    amended_from_id: Mapped[uuid.UUID | None] = mapped_column(
        UUID(as_uuid=True),
        ForeignKey("perf.performance_contract.contract_id"),
        nullable=True,
    )
    amendment_reason: Mapped[str | None] = mapped_column(Text, nullable=True)

    created_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True), server_default=func.now()
    )
    updated_at: Mapped[datetime | None] = mapped_column(
        DateTime(timezone=True), onupdate=func.now(), nullable=True
    )

    # Relationships
    employee: Mapped["Employee"] = relationship(  # noqa: F821
        "Employee", foreign_keys=[employee_id]
    )
    supervisor: Mapped["Employee"] = relationship(  # noqa: F821
        "Employee", foreign_keys=[supervisor_id]
    )
    countersigner: Mapped["Employee | None"] = relationship(  # noqa: F821
        "Employee", foreign_keys=[countersigner_id]
    )
    cycle: Mapped["AppraisalCycle"] = relationship("AppraisalCycle")  # noqa: F821
    amended_from: Mapped["PerformanceContract | None"] = relationship(
        "PerformanceContract", remote_side=[contract_id]
    )
```

from decimal import Decimal

import pytest

from app.services.people.perf.scoring_engine import OHCSFScoringEngine

engine = OHCSFScoringEngine()


class TestRawScore:
    """Step 1: Raw achievement score per KPI."""

    def test_outstanding_achievement(self):
        """Actual meets or exceeds outstanding threshold → 100%."""
        result = engine.calculate_raw_score(
            actual=Decimal("90"),
            thresholds={
                "outstanding": Decimal("85"),
                "excellent": Decimal("80"),
                "good": Decimal("70"),
                "fair": Decimal("60"),
                "poor": Decimal("50"),
            },
        )
        assert result == Decimal("100")

    def test_between_fair_and_good(self):
        """Actual=65, between Fair(60) and Good(70) → 75%."""
        result = engine.calculate_raw_score(
            actual=Decimal("65"),
            thresholds={
                "outstanding": Decimal("85"),
                "excellent": Decimal("80"),
                "good": Decimal("70"),
                "fair": Decimal("60"),
                "poor": Decimal("50"),
            },
        )
        assert result == Decimal("75")

    def test_at_exact_threshold(self):
        """Actual exactly at Good threshold → 80%."""
        result = engine.calculate_raw_score(
            actual=Decimal("70"),
            thresholds={
                "outstanding": Decimal("85"),
                "excellent": Decimal("80"),
                "good": Decimal("70"),
                "fair": Decimal("60"),
                "poor": Decimal("50"),
            },
        )
        assert result == Decimal("80")

    def test_below_poor(self):
        """Actual below Poor threshold → proportional below 60%."""
        result = engine.calculate_raw_score(
            actual=Decimal("25"),
            thresholds={
                "outstanding": Decimal("85"),
                "excellent": Decimal("80"),
                "good": Decimal("70"),
                "fair": Decimal("60"),
                "poor": Decimal("50"),
            },
        )
        assert result < Decimal("60")
        assert result >= Decimal("0")

    def test_inverse_thresholds_lower_is_better(self):

        result = engine.calculate_raw_score(

# app/services/people/perf/scoring_engine.py
"""OHCSF 3-step composite scoring engine."""

from __future__ import annotations

from decimal import ROUND_HALF_UP, Decimal


OHCSF_RATING_SCALE = {
    5: {"label": "Outstanding", "min_pct": Decimal("90")},
    4: {"label": "Excellent", "min_pct": Decimal("80")},
    3: {"label": "Good", "min_pct": Decimal("70")},
    2: {"label": "Fair", "min_pct": Decimal("60")},
    1: {"label": "Poor", "min_pct": Decimal("0")},
}

# Percentage assigned to each rating level
_LEVEL_PCT = {
    "outstanding": Decimal("100"),
    "excellent": Decimal("90"),
    "good": Decimal("80"),
    "fair": Decimal("70"),
    "poor": Decimal("60"),
}

# Ordered from best to worst percentage
_LEVELS_ORDERED = ["outstanding", "excellent", "good", "fair", "poor"]


class OHCSFScoringEngine:
    """Implements the OHCSF composite scoring formula."""

    def calculate_raw_score(
        self,
        actual: Decimal,
        thresholds: dict[str, Decimal],
    ) -> Decimal:
        """
        Step 1: Compare actual achievement against KPI-specific thresholds.

        Supports both ascending (higher=better) and descending (lower=better)
        threshold scales by detecting the direction from the data.

        Returns a percentage 0-100.
        """
        out_t = thresholds["outstanding"]
        poor_t = thresholds["poor"]

        # Detect direction: ascending (higher=better) vs descending (lower=better)
        ascending = out_t > poor_t

        # Build ordered pairs: (threshold_value, percentage) from best to worst
        pairs = [(thresholds[level], _LEVEL_PCT[level]) for level in _LEVELS_ORDERED]

        if ascending:
            # At or above outstanding
            if actual >= pairs[0][0]:
                return Decimal("100")
            # Below poor
            if actual <= pairs[-1][0]:
                # Proportional: (actual / poor_threshold) * 60
                if pairs[-1][0] == 0:
                    return Decimal("0")
                ratio = actual / pairs[-1][0]
                return (ratio * Decimal("60")).quantize(
                    Decimal("0.01"), rounding=ROUND_HALF_UP
                )
            # Interpolate between bands
            for i in range(len(pairs) - 1):
                upper_val, upper_pct = pairs[i]
                lower_val, lower_pct = pairs[i + 1]
                if lower_val <= actual <= upper_val:
                    if upper_val == lower_val:
                        return upper_pct
                    fraction = (actual - lower_val) / (upper_val - lower_val)
                    result = lower_pct + fraction * (upper_pct - lower_pct)
                    return result.quantize(
                        Decimal("0.01"), rounding=ROUND_HALF_UP
                    )
        else:
            # Descending: lower is better (e.g., error rate, turnaround time)
            if actual <= pairs[0][0]:
                return Decimal("100")
            if actual >= pairs[-1][0]:
                if pairs[-1][0] == 0:
                    return Decimal("0")
                ratio = pairs[-1][0] / actual
                return (ratio * Decimal("60")).quantize(
                    Decimal("0.01"), rounding=ROUND_HALF_UP
                )
            for i in range(len(pairs) - 1):
                upper_val, upper_pct = pairs[i]  # lower number = better
                lower_val, lower_pct = pairs[i + 1]  # higher number = worse
                if upper_val <= actual <= lower_val:
                    if lower_val == upper_val:
                        return upper_pct
                    fraction = (lower_val - actual) / (lower_val - upper_val)
                    result = lower_pct + fraction * (upper_pct - lower_pct)
                    return result.quantize(
                        Decimal("0.01"), rounding=ROUND_HALF_UP
                    )

        return Decimal("0")

    def calculate_weighted_score(
        self, raw_score_pct: Decimal, weight: Decimal
    ) -> Decimal:
        """Step 2: raw_score_percentage × weight."""
        return (raw_score_pct * weight).quantize(
            Decimal("0.01"), rounding=ROUND_HALF_UP
        )

    def calculate_composite(self, weighted_scores: list[Decimal]) -> Decimal:
        """Step 3: Sum all weighted raw scores."""
        return sum(weighted_scores, Decimal("0")).quantize(
            Decimal("0.01"), rounding=ROUND_HALF_UP
        )

    def calculate_appraisal_final(
        self,
        objective_composite: Decimal,
        competency_score: Decimal,
        process_score: Decimal,
    ) -> Decimal:
        """Final: objectives(70%) + competencies(20%) + processes(10%)."""
        total = objective_composite + competency_score + process_score
        return total.quantize(Decimal("0.01"), rounding=ROUND_HALF_UP)

    def score_to_rating(self, composite_pct: Decimal) -> tuple[int, str]:
        """Map composite percentage to OHCSF rating (5=Outstanding → 1=Poor)."""
        for rating in sorted(OHCSF_RATING_SCALE.keys(), reverse=True):
            if composite_pct >= OHCSF_RATING_SCALE[rating]["min_pct"]:
                return rating, OHCSF_RATING_SCALE[rating]["label"]
        return 1, "Poor"
```

# app/services/people/perf/contract_service.py
"""Performance Contract Service — OHCSF PMS."""

from __future__ import annotations

import logging
from datetime import date
from typing import TYPE_CHECKING
from uuid import UUID

from sqlalchemy import func, select
from sqlalchemy.orm import Session

from app.models.people.perf.performance_contract import PerformanceContract
from app.models.people.perf.pms_enums import ContractStatus, ContractType
from app.services.common import PaginatedResult, PaginationParams

if TYPE_CHECKING:
    from app.web.deps import WebAuthContext

logger = logging.getLogger(__name__)


class ContractServiceError(Exception):
    pass


class ContractNotFoundError(ContractServiceError):
    def __init__(self, contract_id: UUID):
        super().__init__(f"Performance contract {contract_id} not found")


class ContractValidationError(ContractServiceError):
    pass


class ContractStatusError(ContractServiceError):
    def __init__(self, current: str, target: str):
        super().__init__(f"Cannot transition from {current} to {target}")


class PerformanceContractService:
    """Manages performance contracts lifecycle."""

    def __init__(self, db: Session, ctx: WebAuthContext | None = None) -> None:
        self.db = db
        self.ctx = ctx

    def _validate_objectives(self, objectives: list[dict]) -> None:
        if not (3 <= len(objectives) <= 7):
            raise ContractValidationError(
                f"Contract must have 3 to 7 objectives, got {len(objectives)}"
            )
        total_weight = sum(obj.get("weight", 0) for obj in objectives)
        if total_weight != 70:
            raise ContractValidationError(
                f"Objective weights must sum to 70, got {total_weight}"
            )

    def _validate_competency_selections(self, competencies: list[dict]) -> None:
        dev_focus_count = sum(
            1 for c in competencies if c.get("is_development_focus")
        )
        if dev_focus_count != 3:
            raise ContractValidationError(
                f"Must select exactly 3 competencies for development focus, got {dev_focus_count}"
            )

    def get_contract(self, org_id: UUID, contract_id: UUID) -> PerformanceContract:
        contract = self.db.scalar(
            select(PerformanceContract).where(
                PerformanceContract.contract_id == contract_id,
                PerformanceContract.organization_id == org_id,
            )
        )
        if not contract:
            raise ContractNotFoundError(contract_id)
        return contract

    def list_contracts(
        self,
        org_id: UUID,
        *,
        cycle_id: UUID | None = None,
        employee_id: UUID | None = None,
        status: ContractStatus | None = None,
        search: str | None = None,
        pagination: PaginationParams | None = None,
    ) -> PaginatedResult[PerformanceContract]:
        query = select(PerformanceContract).where(
            PerformanceContract.organization_id == org_id
        )
        if cycle_id:
            query = query.where(PerformanceContract.cycle_id == cycle_id)
        if employee_id:
            query = query.where(PerformanceContract.employee_id == employee_id)
        if status:
            query = query.where(PerformanceContract.status == status)
        if search:
            query = query.where(
                PerformanceContract.contract_code.ilike(f"%{search}%")
            )

        query = query.order_by(PerformanceContract.created_at.desc())
        count_query = select(func.count()).select_from(query.subquery())
        total = self.db.scalar(count_query) or 0

        if pagination:
            query = query.offset(pagination.offset).limit(pagination.limit)

        items = list(self.db.scalars(query).all())
        return PaginatedResult(
            items=items,
            total=total,
            offset=pagination.offset if pagination else 0,
            limit=pagination.limit if pagination else len(items),
        )

    def create_contract(
        self,
        org_id: UUID,
        *,
        cycle_id: UUID,
        employee_id: UUID,
        supervisor_id: UUID,
        contract_code: str,
        contract_type: ContractType,
        objectives: list[dict],
        competency_ids: list[dict] | None = None,
        development_plan: str | None = None,
    ) -> PerformanceContract:
        self._validate_objectives(objectives)
        if competency_ids:
            self._validate_competency_selections(competency_ids)

        contract = PerformanceContract(
            organization_id=org_id,
            cycle_id=cycle_id,
            employee_id=employee_id,
            supervisor_id=supervisor_id,
            contract_code=contract_code,
            contract_type=contract_type,
            objectives=objectives,
            competency_ids=competency_ids,
            development_plan=development_plan,
        )
        self.db.add(contract)
        self.db.flush()
        logger.info("Created performance contract: %s", contract.contract_code)
        return contract

    def sign_employee(self, org_id: UUID, contract_id: UUID) -> PerformanceContract:
        contract = self.get_contract(org_id, contract_id)
        contract.employee_signed_date = date.today()
        if contract.supervisor_signed_date:
            contract.status = ContractStatus.ACTIVE
        else:
            contract.status = ContractStatus.PENDING_SIGNATURE
        self.db.flush()
        return contract

    def sign_supervisor(self, org_id: UUID, contract_id: UUID) -> PerformanceContract:
        contract = self.get_contract(org_id, contract_id)
        contract.supervisor_signed_date = date.today()
        if contract.employee_signed_date:
            contract.status = ContractStatus.ACTIVE
        else:
            contract.status = ContractStatus.PENDING_SIGNATURE
        self.db.flush()
        return contract

    def countersign(
        self, org_id: UUID, contract_id: UUID, countersigner_id: UUID
    ) -> PerformanceContract:
        contract = self.get_contract(org_id, contract_id)
        contract.countersigner_id = countersigner_id
        contract.countersigner_date = date.today()
        if not contract.employee_signed_date:
            contract.status = ContractStatus.ACTIVE
        self.db.flush()
        return contract

    def amend_contract(
        self,
        org_id: UUID,
        contract_id: UUID,
        *,
        new_objectives: list[dict],
        amendment_reason: str,
        competency_ids: list[dict] | None = None,
    ) -> PerformanceContract:
        original = self.get_contract(org_id, contract_id)
        self._validate_objectives(new_objectives)

        next_code = original.contract_code + "-A"
        new_contract = PerformanceContract(
            organization_id=org_id,
            cycle_id=original.cycle_id,
            employee_id=original.employee_id,
            supervisor_id=original.supervisor_id,
            contract_code=next_code,
            contract_type=original.contract_type,
            objectives=new_objectives,
            competency_ids=competency_ids or original.competency_ids,
            development_plan=original.development_plan,
            amended_from_id=original.contract_id,
            amendment_reason=amendment_reason,
        )
        original.status = ContractStatus.AMENDED
        self.db.add(new_contract)
        self.db.flush()
        logger.info("Amended contract %s → %s", original.contract_code, next_code)
        return new_contract
